function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = JSON.parse(e.postData.contents); // ترتيب الأعمدة: الوقت | النتيجة | رقم المحاولة | إحصائيات الأسئلة 1-15 var row = [new Date(), data.score + " / 15", "المحاولة رقم: " + data.attempt]; data.qStats.forEach(function(res) { row.push(res); }); sheet.appendRow(row); return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT); }
